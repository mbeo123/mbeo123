while true do
-- [[ CẤU HÌNH ]]
local UNIT_UUID = "a7a9021e-6764-4324-9bcb-ced195c2b56d"
local SPAWN_POSITION = CFrame.new(-277.50067138671875, 63.57316970825195, -123.35144805908203) * CFrame.Angles(-0, 0, -0)

local Remotes = game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("client_server")
local SpawnRemote = Remotes:WaitForChild("unit_spawn")
local UpgradeRemote = Remotes:WaitForChild("unit_upgrade")
local ReplayRemote = Remotes:WaitForChild("teleport_replay")

-- GỐC VẤN ĐỀ: Hai biến này lưu trữ mốc ID. Nó sẽ TĂNG TIẾN MÃI MÃI, replay không bị reset!
local startID = 1
local maxID = 20 -- Trận đầu tiên, check từ 1 đến 20

-- [[ 1. VÒNG LẶP ĐẶT VÀ SPAM NÂNG CẤP ĐUỔI THEO ID SERVER ]]
task.spawn(function()
    while true do
        -- Lệnh đặt Unit
        pcall(function()
            SpawnRemote:InvokeServer(UNIT_UUID, SPAWN_POSITION)
        end)
        task.wait(0.5) -- Chờ game nhận lệnh đặt

        -- Nâng cấp "Rải thảm" theo vùng ID động
        -- Ví dụ: Trận 1 quét 1->20. Trận 2 quét 15->35. Trận 3 quét 30->50... 
        for i = startID, maxID do
            pcall(function()
                UpgradeRemote:InvokeServer(tostring(i))
            end)
            task.wait(0.02) -- Tốc độ quét cực nhanh (20ms/ID)
        end
        
        -- CƠ CHẾ ĐẨY MỐC: Mỗi lần đặt+nâng cấp xong, ta tịnh tiến vùng quét lên để đón đầu ID mới của server
        -- Giữ lại một khoảng lùi nhỏ (trừ 5) đề phòng game bị delay hoặc đặt trượt
        startID = math.max(1, maxID - 5) 
        maxID = startID + 20 -- Nới rộng vùng quét thêm 20 ID cho lượt tiếp theo
        
        task.wait(1.5) -- Nghỉ một chút rồi lặp lại
    end
end)

-- [[ 2. VÒNG LẶP REPLAY ]]
task.spawn(function()
    while true do
        pcall(function()
            ReplayRemote:InvokeServer()
        end)
        task.wait(5)
    end
end)
task.wait(1)
end
